/**
 * Header class for Content-Location.
 *
 * <h3>RFC822</h3>
 *
 * <pre>
 * content-location := "Content-Location:" [CFWS] URI [CFWS]
 *
 * URI := absoluteURI | relativeURI
 * </pre>
 *
 * <hr>
 *
 * @package BLW\MIME
 * @api BLW
 * @since 1.0.0
 * @author mAsT3RpEE <wotsyula@mast3rpee.tk>
 */
final class ReplyTo extends \BLW\Type\MIME\AHeader
{

    /**
     * Constructor
     *
     * @throws \BLW\Model\InvalidArgumentException If <code>$AddressList</code> is invalid.
     *
     * @param \BLW\Type\IURI $Location
     *            Value of Content-Location header
     */
    public function __construct(IContainer $AddressList)
    {

        // 1. Header type
        $this->_Type = 'Reply-To';

        // 2. Header value
        $this->_Value = $this->parseAddressList($AddressList);

        // Check value
        if (! $this->_Value)
            throw new InvalidArgumentException(0);
    }
}

// @codeCoverageIgnoreStart
